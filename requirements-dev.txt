# CPC local DEV tooling — macOS dev machine ONLY. NOT for console nodes.

#   python3 -m venv .venv
#   .venv/bin/pip install -r requirements-dev.txt
#
# start-pluto.sh auto-detects .venv. For the CLI use .venv/bin/python (or
# `source .venv/bin/activate` once, then plain `python`).
